 void sort(int *arr,int n)//选择排序

{

    for (int i = 0; i < n; i++)

    {

        int maxIndex = i; //最大值下标

        for (int j = i + 1; j < n; j++)

        {         

            if(arr[maxIndex]<arr[j])//降序

                maxIndex = j;

        }

        //将最大元素和第i个元素交换

        if (i != maxIndex)

        { //交换条件

            int temp = arr[maxIndex];

            arr[maxIndex] = arr[i];

            arr[i] = temp;

        }

    }

}
